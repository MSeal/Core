15 /*
16  * TSVector implements a boost::thread safe version of a vector.
17  * The implementation uses a wrapped vector for all calls. A
18  * limited number of base calls are available without retrieving
19  * a locked reference of the vector.
20  */
21 template <typename T, typename Alloc = std::allocator<T> >
22 class TSVector : public TSReadWriteWrapper<std::vector<T, Alloc> > {
23 private:
24  // Class renaming for readability
25  typedef std::vector<T, Alloc> VectorType;
26  typedef boost::shared_ptr<VectorType> VectorTypePtr;
27  typedef TSReadWriteWrapper<VectorType> Vector;
28  typedef typename Vector::ScopedLock ScopedLock;
29  typedef typename Vector::SharedScopedLock SharedScopedLock;
30 
31  // Assignment operator is not made, because all lockable objects
32  // are non-copyable.
33 
34 public:
35  explicit TSVector(int priority = 0) : Vector(priority) {}
36  explicit TSVector(VectorType& other, int priority = 0) : Vector(priority) {
37  this->wrapped = VectorTypePtr(VectorType(other));
38  }
39  explicit TSVector(TSVector& other) : Vector(other.getPriority()) {
40  other.lock();
41  this->wrapped = VectorTypePtr(VectorType(other.wrapped));
42  other.unlock();
43  }
44  explicit TSVector(TSVector& other, int priority) : Vector(priority) {
45  other.lock();
46  this->wrapped = VectorTypePtr(VectorType(other.wrapped));
47  other.unlock();
48  }
49  template <typename InputIterator>
50  TSVector(InputIterator first, InputIterator last, int priority = 0) : Vector(priority) {
51  this->wrapped = VectorTypePtr(VectorType(first, last));
52  }
