141  // Extracts the LogLevel from the boost exception and calls logMessage
142  void logException(const boost::exception &x) {
143  LogLevel loglevel;
144  switch(getExceptionSeverity(x)) {
145  case EXCEP_SEVERITY_NOTICE:
146  loglevel = LOG_INFO;
147  break;
148  case EXCEP_SEVERITY_WARNING:
149  loglevel = LOG_WARNING;
150  break;
151  case EXCEP_SEVERITY_ERROR:
152  default:
153  loglevel = LOG_ERROR;
154  break;
155  }
156  logMessage(loglevel, x);
157  }
